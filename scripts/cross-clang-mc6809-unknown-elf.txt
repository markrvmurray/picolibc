# Memory layout (top-to-bottom; $0000 at the top, $FFFF at the
# bottom). The emulated machine has no real flash — the whole
# $0000-$FFCF range is one writable RAM, with USim's IO slab + vector
# ROM at $FFD0-$FFFF. picolibc's template still requires two MEMORY
# regions ('flash' and 'ram'), so we keep a 16-byte 'flash' stub at
# $FFC0 (just above the IO slab) holding only the .init reset entry,
# and let 'ram' cover everything else. picolibc_ld_data is patched
# upstream in meson.build to put .text + .rodata + .data + .bss +
# heap + stack ALL into ram, so the full 64-KB-minus-IO is available
# to the program no matter the section mix.
#   ram    $0100-$FFBF  (~63.7 KB) — text + rodata + data + bss + heap + stack
#   flash  $FFC0-$FFCF  (16 bytes) — .init only (reset entry stub)
#   IO     $FFD0-$FFEF              — USim ACIA + halt + tracectl
#   vrom   $FFF0-$FFFF              — vector ROM
default_flash_addr = '0xFFC0'
default_flash_size = '0x0010'
default_ram_addr   = '0x0100'
default_ram_size   = '0xFEC0'
