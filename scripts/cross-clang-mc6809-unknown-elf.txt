# Memory layout (top-to-bottom; $0000 at the top, $FFFF at the
# bottom). The emulated machine has no real flash — the whole
# $0000-$FFCF range is one writable RAM, with USim's IO slab + vector
# ROM at $FFD0-$FFFF. picolibc's template still requires two MEMORY
# regions ('flash' and 'ram'), so we keep a 16-byte 'flash' stub at
# $FFC0 (just above the IO slab) holding only the .init reset entry,
# and let 'ram' cover everything else. picolibc_ld_data is patched
# upstream in meson.build to put .text + .rodata + .data + .bss +
# heap + stack ALL into ram, so the full 64-KB-minus-IO is available
# to the program no matter the section mix.
#   ram    $0100-$FFBF  (~63.7 KB) — text + rodata + data + bss + heap + stack
#   flash  $FFC0-$FFCF  (16 bytes) — .init only (reset entry stub)
#   IO     $FFD0-$FFEF              — USim ACIA + halt + tracectl
#   vrom   $FFF0-$FFFF              — vector ROM
default_flash_addr = '0xFFC0'
default_flash_size = '0x0010'
default_ram_addr   = '0x0100'
default_ram_size   = '0xFEC0'

# Bug #192: direct-page support. The compiler's imaginary registers
# (RC0..RC15, RS0..RS7) claim $00-$1F via .set directives in crt0.c,
# plus __scratch at $20 (1 byte). User __directpage globals fill
# $21-$FF (223 bytes); link-time ASSERT catches overflow. The
# DIRECTPAGE_MEMORY string is appended to the MEMORY block in
# picolibc.ld.in; DIRECTPAGE_BLOCK is appended after the .heap
# SECTIONS entry.
directpage_memory = '''
	directpage (rw!x) :
		ORIGIN = 0x0021,
		LENGTH = 0x00DF'''

directpage_block = '''
	.dp_data : {
		PROVIDE(__dp_data_start = .);
		*(.dp.data .dp.data.*)
		PROVIDE(__dp_data_end = .);
		*(.dp.rodata .dp.rodata.*)
	} > directpage :ram_init

	.dp_bss (NOLOAD) : {
		PROVIDE(__dp_bss_start = .);
		*(.dp.bss .dp.bss.*)
		PROVIDE(__dp_bss_end = .);
	} > directpage :NONE

	PROVIDE(__dp_data_size = __dp_data_end - __dp_data_start);
	PROVIDE(__dp_data_source = ADDR(.dp_data));
	PROVIDE(__dp_bss_size = __dp_bss_end - __dp_bss_start);

	ASSERT(__dp_bss_end <= 0x100,
		"Direct page overflow: .dp.* sections exceed $00-$FF.")'''
